% ==========================================================
% Figure fitting helpers (SNmono-safe)
% ==========================================================
% Maximum figure dimensions: allows figures to be 15% wider than linewidth
% and up to 90% of page height. These values can be adjusted per-figure
% if needed by redefining \LLMFigMaxWidth or \LLMFigMaxHeight locally.
\newcommand{\LLMFigMaxWidth}{1.15\linewidth}
\newcommand{\LLMFigMaxHeight}{0.9\textheight}

% A wrapper that scales DOWN to fit the Springer textblock,
% and prevents overly-tall diagrams from spilling off the page.
% Note: adjustbox scales content proportionally, so increasing TikZ sizes
% will be visible up to these maximum constraints.
\newenvironment{llmfigbox}{%
  \begin{adjustbox}{max width=\LLMFigMaxWidth, max totalheight=\LLMFigMaxHeight, center}%
}{%
  \end{adjustbox}%
}

% A wrapper that centers content without scaling constraints.
% Use this when you want full control over figure sizing.
\newenvironment{llmfigboxnoscale}{%
  \begin{adjustbox}{center}%
}{%
  \end{adjustbox}%
}

% ==========================================================
% Code Listings: minted + tcolorbox
% ==========================================================
% Produces a tcolorbox container that matches the visual language of
% \BestPracticeBox and \ChecklistBox. Code is typeset by minted (real
% syntax highlighting via Pygments). Numbering is chapter-scoped:
% "Listing 1.1", "Listing 7.3", etc.
%
% Usage:
%   \begin{codebox}{Caption stating the reader takeaway.}{lst:chXX_my_label}
%   \begin{minted}{json}      % or python, yaml, bash, etc.
%   ...code...
%   \end{minted}
%   \end{codebox}
%
% For listings without a label, pass an empty third argument: {}
% ==========================================================

% Default minted setup -- grayscale-safe, Springer-friendly font sizing.
\setminted{
  style=bw,
  fontsize=\footnotesize,
  breaklines=true,
  breakanywhere=false,
  breaksymbolleft=\textcolor{black!40}{\tiny\ensuremath{\hookrightarrow}},
  numbers=left,
  numbersep=8pt,
  xleftmargin=1.6em,
  framesep=2mm,
  tabsize=2,
  autogobble=true,
}

% Chapter-scoped counter so "Listing X.Y" matches figure/table conventions.
\newcounter{codelisting}[chapter]
\renewcommand{\thecodelisting}{\thechapter.\arabic{codelisting}}

% The container. Matches \BestPracticeBox / \ChecklistBox visual language.
% Args: #1 = caption (reader takeaway), #2 = label (e.g., lst:ch01_foo)
\newtcolorbox{codebox}[2]{%
  enhanced, breakable,
  colback=black!2,
  colframe=black!55,
  colbacktitle=black!10,
  coltitle=black,
  fonttitle=\bfseries\small,
  title={\refstepcounter{codelisting}\label{#2}Listing \thecodelisting},
  boxrule=0.4pt,
  arc=2pt,
  left=2mm, right=2mm, top=1mm, bottom=1mm,
  before skip=10pt, after skip=10pt,
  attach boxed title to top left={xshift=2mm, yshift=-2mm},
  boxed title style={boxrule=0pt, arc=1pt},
  after upper={\par\smallskip\noindent\itshape\footnotesize #1\par},
}

% ==========================================================
% Figure style contract for TikZ + PGFPlots
% ==========================================================
% Single source of truth for diagram and chart aesthetics.
% Every figure in the book should use these styles to ensure
% grayscale safety, consistent typography, print-safe line
% weights, and Springer textblock fit.
% ==========================================================

% Brand-neutral palette: each color has a distinct grayscale value
% so figures remain legible in print without relying on hue alone.
% Sources:
%   - Hue separation: ~30 deg between adjacent slots
%   - Lightness separation: at least 12 L* between adjacent slots
\definecolor{llmblue}{RGB}{44,102,146}     % L* ~ 41
\definecolor{llmorange}{RGB}{201,111,29}   % L* ~ 56
\definecolor{llmgreen}{RGB}{34,139,96}     % L* ~ 51
\definecolor{llmpurple}{RGB}{123,88,163}   % L* ~ 44
\definecolor{llmgray}{RGB}{90,90,90}       % L* ~ 38

% Diagram styles. Use these in every \begin{tikzpicture} so node
% shapes, line weights, and arrow heads are uniform book-wide.
\tikzset{
  llm/box/.style={
    draw=black!70, line width=0.6pt, rounded corners=3pt,
    align=center, inner sep=4pt, minimum height=9mm, font=\small
  },
  llm/box/blue/.style   ={llm/box, fill=llmblue!12,   draw=llmblue!70},
  llm/box/orange/.style ={llm/box, fill=llmorange!12, draw=llmorange!70},
  llm/box/green/.style  ={llm/box, fill=llmgreen!12,  draw=llmgreen!70},
  llm/box/purple/.style ={llm/box, fill=llmpurple!12, draw=llmpurple!70},
  llm/box/gray/.style   ={llm/box, fill=black!4,      draw=black!50},
  llm/arrow/.style={-{Latex[length=2.2mm,width=1.6mm]}, line width=0.7pt, color=black!75},
  llm/arrow/dashed/.style={llm/arrow, dashed, dash pattern=on 3pt off 2pt, color=black!60},
  llm/lane/.style={draw=black!40, rounded corners=4pt, dashed, inner sep=8pt},
  llm/label/.style={font=\footnotesize, align=left, text=black!75},
}

% PGFPlots defaults. Apply with [llm/chart] inside \begin{axis}[...].
\pgfplotsset{

    height=6.5cm,
    font=\small,
    xlabel style={font=\small\bfseries},
    ylabel style={font=\small\bfseries},
    ticklabel style={font=\footnotesize},
    legend style={

      cells={anchor=west}, at={(0.02,0.97)}, anchor=north west
    },

    axis line style={line width=0.6pt, black!75},
    tick align=outside,
    tick style={black!60},

  },
  % Cycle list with print-safe lightness separation; uses dash patterns
  % to remain distinguishable in grayscale.
  llm/cycle/.style={
    cycle list={
      {color=llmblue,   solid,                          mark=*,        mark size=1.5pt},
      {color=llmorange, dashed, dash pattern=on 4pt off 2pt, mark=square*, mark size=1.5pt},
      {color=llmgreen,  dotted, dash pattern=on 1pt off 1.5pt, mark=triangle*, mark size=1.8pt},
      {color=llmpurple, dashdotted,                     mark=diamond*, mark size=1.8pt},
      {color=llmgray,   loosely dashed,                 mark=x,        mark size=1.8pt},
    }
  }
}
